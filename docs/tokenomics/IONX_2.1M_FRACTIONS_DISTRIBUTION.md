## üî¢ Mathematical Formula

### Universal Per-Fraction Calculator

```javascript
function calculateLifetimeIONX(fractions) {
  const IONX_PER_FRACTION = 2433.33;
  return fractions * IONX_PER_FRACTION;
}

function calculateDailyIONX(fractions, epoch) {
  const INITIAL_DAILY = 1_000_000;
  const VALIDATOR_SHARE = 0.70;
  const TOTAL_FRACTIONS = 2_100_000;
  
  const dailyEmission = INITIAL_DAILY / Math.pow(2, epoch);
  const validatorRewards = dailyEmission * VALIDATOR_SHARE;
  const perFraction = validatorRewards / TOTAL_FRACTIONS;
  
  return fractions * perFraction;
}

// Examples:
console.log(calculateLifetimeIONX(1));      // 2,433.33 IONX
console.log(calculateLifetimeIONX(100));    // 243,333 IONX
console.log(calculateLifetimeIONX(1000));   // 2,433,333 IONX

console.log(calculateDailyIONX(100, 0));    // 33.33 IONX/day (Year 1-2)
console.log(calculateDailyIONX(100, 1));    // 16.67 IONX/day (Year 3-4)
console.log(calculateDailyIONX(100, 2));    // 8.33 IONX/day (Year 5-6)
```
