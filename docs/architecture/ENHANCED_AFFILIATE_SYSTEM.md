```solidity
// Add to ValidatorFractionNFT.sol

struct AffiliateStats {
    uint256 totalReferrals;
    uint256 totalVolume;
    uint256 totalEarnings;
    AffiliateRank rank;
    mapping(uint256 => bool) achievementsEarned;
}

mapping(address => AffiliateStats) public affiliateStats;

// Achievement events
event AchievementEarned(address indexed affiliate, uint256 achievementId, uint256 reward);
event RankUpgraded(address indexed affiliate, AffiliateRank newRank, uint256 bonus);
event TargetCompleted(address indexed affiliate, string targetType, uint256 bonus);

// Achievement checking
function checkAchievements(address affiliate) internal {
    AffiliateStats storage stats = affiliateStats[affiliate];
    
    // First referral
    if (stats.totalReferrals == 1 && !stats.achievementsEarned[1]) {
        stats.achievementsEarned[1] = true;
        ionxToken.transfer(affiliate, 50 * 10**18);
        emit AchievementEarned(affiliate, 1, 50 * 10**18);
    }
    
    // 10 referrals
    if (stats.totalReferrals >= 10 && !stats.achievementsEarned[2]) {
        stats.achievementsEarned[2] = true;
        ionxToken.transfer(affiliate, 500 * 10**18);
        emit AchievementEarned(affiliate, 2, 500 * 10**18);
    }
    
    // More achievements...
}

// Rank upgrade
function checkRankUpgrade(address affiliate) internal {
    AffiliateStats storage stats = affiliateStats[affiliate];
    AffiliateRank newRank = calculateRank(stats.totalReferrals, stats.totalVolume);
    
    if (newRank > stats.rank) {
        stats.rank = newRank;
        uint256 bonus = getRankUpBonus(newRank);
        ionxToken.transfer(affiliate, bonus);
        emit RankUpgraded(affiliate, newRank, bonus);
    }
}
```
